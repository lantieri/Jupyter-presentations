def chgtNuages(ptsFilt,G):
    n_clusters=len(G)
    gr=[[]for g in G]   
    for pt in ptsFilt :
        dMin = d(pt,G[0])
        iMin = 0
        for i in range(1,n_clusters) :
            if d(pt,G[i])< dMin:
                dMin = d(pt,G[i])
                iMin = i
        gr[iMin].append(pt)
    return gr
